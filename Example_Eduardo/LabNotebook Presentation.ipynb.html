For doing math¶

  • Recursive division: Having that each child defines a new unit cell, the $i$-th child will have boundaries ${\bf l}_i$ and ${\bf u}_i$ given as ${\bf l}_i = {\bf l} + \Delta{\bf x}( \text{cell}_i - \mathbb{1} )$ and ${\bf u}_i = {\bf l} + \Delta{\bf x}( \text{cell}_i )$, with ${\bf l}$ and ${\bf u}$ the boundaries of its parent, $\mathbb{1}\in\mathbb{R}^d$ the unitary vector $\mathbb{1}=(1,\dots,1)$, and

\begin{align} \Delta{\bf x} : & \mathbb{N}^d\rightarrow\mathbb{R}^d \\ & \text{cell} \rightarrow ({\bf u} - {\bf l}) .* \Big( \sum\limits_{k=0}^{(cell)_1} (\ell_1)_k, \dots, \sum\limits_{k=0}^{(cell)_d} (\ell_d)_k\Big) \end{align}

with $(\ell_i)_0 = 0$ and $.*$ denoting the element-to-element product.

  • Child cells: A unit cell has $n_c$ child cells with $n_c=\prod\limits_{i=1}^{d} n_i $. They are numbered in indices from 1 to $n_c$ organized in the matrix of subdivision as follows:

\begin{equation} \text{cell}_i = \begin{bmatrix} [(i-1)\%n_1]+1 & \text{floor}(\frac{i-1}{n_1})\%n_2+1 & \text{floor}(\frac{i-1}{n_1 n_2})\%n_3+1 & \cdots & \text{floor}(\frac{i-1}{n_1\cdots n_{d-1}})\%n_d+1 \end{bmatrix} \end{equation}

or

\begin{equation} \big( \text{cell}_i \big)_j = \text{floor}\Big(\frac{i-1}{\tiny \prod\limits_{k=1}^{j-1} n_k}\Big)\%n_j+1 \qquad \text{, with } n_0 = 1 \end{equation}

function calc_cell(i, n, d)
    cell = zeros(Array{Int64}(d))
    for j in 1:d
        if j==1
            cell[j] = Int(((i-1)%n[j]) + 1)
        else
            prod = 1
            for k in 1:j-1
                prod = prod*n[k]
            end
            cell[j] = Int(( floor((i-1)/prod) )%n[j] + 1)
        end
    end
    return cell
end

function calc_b(i, l, u, d)
    b = zeros(d)
    for j in 1:d
        binary = floor((i-1)/2^(j-1))%2+1
        if binary==1
            b[j] = l[j]
        elseif binary==2
            b[j] = u[j]
        else
            error("Logic error! binary not in [1,2]")
        end
    end
    return b
end

function delta_x(cell, l, u, L::Array{Array{Float64,1},1}, d)
#     _L = L/sum(L)    # Just making sure they are normalized lengths
    _L = []
    for i in 1:d
        push!(_L, L[i]/sum(L[i]))
    end
    aux = zeros(d)
    for j in 1:d
        sm = 0
        for k in 1:cell[j]
            sm += _L[j][k]
        end
        aux[j] = sm
    end
    return (u-l).*aux
end

# ORTHOGONAL SYSTEMS (See https://en.wikipedia.org/wiki/Orthogonal_coordinates)
function cylindrical(X)
    r, theta, z = X
    return Float64[ r*cos(theta*pi/180), r*sin(theta*pi/180), z ]
end
function spherical(X)
    r, theta, phi = X
    return Float64[ r*sin(theta*pi/180)*cos(phi*pi/180),
                    r*sin(theta*pi/180)*sin(phi*pi/180),
                    r*cos(theta*pi/180) ]
end
function parabolic(X)
    u,v,z = X
    return Float64[ 1/2*(u^2-v^2), u*v, z ]
end
function cylindrical2D(X)
    r, theta = X[1], X[2]
    return Float64[ r*cos(theta*pi/180),  r*sin(theta*pi/180)]
end
function cartesian(X)
    return X
end
function conv2threeD(X)
    X3d = zeros(3)
    for i in 1:3
        if i<=size(X)[1]
            X3d[i]=X[i]
        end
    end
    return X3d
end

# PROBABILITY DISTRIBUTIONS
function uniform(x)
  return x
end
function weibull(x; lambda=1, k=1.5, scale=2, norm=1.5)
  r = x*scale
  return k/lambda*(r/lambda)^(k-1)*exp(-(r/lambda)^k)/norm
end
function normal(x; mu=0.5, sigmasqr=1.0)
  res = 1/(2*pi*sigmasqr)^0.5 * exp(-(x-mu)^2/(2*sigmasqr))
  return res
end
function random(x)
  return rand()
end
function circle(x; r=0.5, scale=0.5)
  side = (rand()>0.5?-1:1)
  val = ( sqrt(  (scale*r)^2 - (scale*(x-0.5))^2  ) + 0.5 )
  return (side==-1 ? 1-val : val)
end

space_transf = cartesian            # Space transformation
part_distr = circle                # Distribution of the particles
part_scttr = 0.5                   # Scatter of particles distribution

# FMM PARAMETERS
nf_max = 5                          # Levels
d = 2                               # Dimensions
# Cartesian
l = [1.0 for i in 1:d]              # Lower bound

# u = [1.0, 1.0, 1.0]

# u = [1.0, 1.0, 1.0]              
ndivs = 2                           # Number of divisions

# VPM PARAMETERS

Vinf(X,t) = [0.0,0.0,0.0]

# Generates the particles in the VPM
for i in 1:n_particles
    X = zeros(3)
    for j in 1:3
        if j<=d
            if j==1
              X[j] = (u[j]+l[j])/2 + 1.0*(rand()-0.5)*(u[j]-l[j])
            else
              prev_x = (X[j-1] - l[j-1]) / (u[j-1]-l[j-1])
              x = l[j] + ((u[j]-l[j])*part_distr(prev_x))*(1 + (rand()-0.5)*part_scttr)
              x = x>=u[j] ? u[j]-0.00001 : (x<l[j] ? l[j]+0.000001 : x)
              X[j] = x
            end
        else
            X[j] = 0.0
        end
    end
    vpm.addparticle(p_field, vcat(X, zeros(4)))
end

for i in 0:vpm.get_np(p_field)

    if i==0
        nothing
    else
